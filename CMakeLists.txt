cmake_minimum_required (VERSION 3.21)

if (POLICY CMP0141)
  cmake_policy(SET CMP0141 NEW)
  set(CMAKE_MSVC_DEBUG_INFORMATION_FORMAT "$<IF:$<AND:$<C_COMPILER_ID:MSVC>,$<CXX_COMPILER_ID:MSVC>>,$<$<CONFIG:Debug,RelWithDebInfo>:EditAndContinue>,$<$<CONFIG:Debug,RelWithDebInfo>:ProgramDatabase>>")
endif()

project ("epix_engine.cmake")

set(CMAKE_CXX_STANDARD 20)

if (MSVC)
  add_compile_options(/bigobj)
else ()
  add_compile_options(-Wa,-mbig-obj)
endif ()

#find third-party if possible
find_package (OpenGL REQUIRED)

#sub directories

add_library(BSThreadPool INTERFACE)
target_include_directories(BSThreadPool INTERFACE ${CMAKE_CURRENT_SOURCE_DIR}/libs/thread-pool/include)

set (EPIX_WGPU_IMPL_DAWN OFF)
set (EPIX_WGPU_IMPL_WGPU_NATIVE ON)

add_subdirectory (libs/entt)
add_subdirectory (libs/glad)
add_subdirectory (libs/glfw)
add_subdirectory (libs/vulkan-headers)
if (WIN32)
  set (VOLK_STATIC_DEFINES VK_USE_PLATFORM_WIN32_KHR)
elseif (UNIX)
  set (VOLK_STATIC_DEFINES VK_USE_PLATFORM_XCB_KHR)
endif ()
add_subdirectory (libs/volk)
add_subdirectory (libs/SPIRV-Cross)
if (EPIX_WGPU_IMPL_DAWN)
  set (DAWN_FETCH_DEPENDENCIES ON)
  set (DAWN_ENABLE_INSTALL ON)
  add_subdirectory (libs/dawn)
  add_library(webgpu ALIAS dawn::webgpu_dawn)
elseif (EPIX_WGPU_IMPL_WGPU_NATIVE)
  set (WEBGPU_BACKEND "wgpu_static")
  add_subdirectory (libs/webgpu)
endif ()
add_subdirectory (libs/glfw_webgpu)
add_subdirectory (libs/spdlog)
add_subdirectory (libs/fmt)
add_subdirectory (libs/pfr)
add_subdirectory (libs/glm)
add_subdirectory (libs/freetype)
add_subdirectory (libs/vma)
add_subdirectory (libs/imgui)
add_subdirectory (libs/box2d)
add_subdirectory (libs/earcut.hpp)
add_subdirectory (libs/tracy)
add_subdirectory (epix_engine)
add_subdirectory (examples)
